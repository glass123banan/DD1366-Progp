# Formell grammatik i BNF för Leona

<program> ::= <expr>
<expr> ::= <instruction> <expr> | ε
<instruction> ::= FORW NUM DOT
               | BACK NUM DOT
               | LEFT NUM DOT
               | RIGHT NUM DOT
               | DOWN DOT
               | UP DOT
               | COLOR HEX DOT
               | REP NUM <rep>
            
<rep> ::= QUOTE <expr> QUOTE | <instruction>


NUM: [0-9]+
DOT: .
QUOTE: "
HEX: # [0-9a-fA-F]{6}
